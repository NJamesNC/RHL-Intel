## Phase 3: Core Files

### `src/index.ts` — Express Entry Point

```typescript
import express from 'express';
import cors from 'cors';
import helmet from 'helmet';
import rateLimit from 'express-rate-limit';
import dotenv from 'dotenv';
import { leadsRouter } from './routes/leads';
import { paymentsRouter } from './routes/payments';

dotenv.config();

const app = express();
const PORT = process.env.PORT || 3000;

app.use(helmet());
app.use(cors({ origin: process.env.FRONTEND_URL || '*', credentials: true }));

const limiter = rateLimit({ windowMs: 15 * 60 * 1000, max: 100 });
app.use('/api/', limiter);

// Raw body for Stripe webhooks — must come BEFORE express.json()
app.use('/api/payments/webhook', express.raw({ type: 'application/json' }));
app.use(express.json({ limit: '10mb' }));

app.use('/api/leads', leadsRouter);
app.use('/api/payments', paymentsRouter);

app.get('/health', (req, res) => res.json({ status: 'ok', timestamp: new Date().toISOString() }));

app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
export default app;
```

### `src/routes/leads.ts` — Lead Capture API

```typescript
import { Router } from 'express';
import { z } from 'zod';
import { db } from '../db';
import { leads } from '../db/schema';

export const leadsRouter = Router();

const leadSchema = z.object({
  name: z.string().min(2).max(100),
  email: z.string().email(),
  phone: z.string().optional(),
  company: z.string().optional(),
  message: z.string().max(1000).optional(),
  source: z.string().optional(),
  utmSource: z.string().optional(),
  utmMedium: z.string().optional(),
  utmCampaign: z.string().optional(),
});

leadsRouter.post('/', async (req, res) => {
  try {
    const data = leadSchema.parse(req.body);
    const [lead] = await db.insert(leads).values(data).returning();

    // Fire to GHL + n8n in parallel — if one fails, the other still lands
    const integrations = [
      process.env.GHL_WEBHOOK_URL,
      process.env.N8N_WEBHOOK_URL,
    ].filter(Boolean);

    await Promise.allSettled(
      integrations.map(url =>
        fetch(url!, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ ...lead, source: 'website_form', timestamp: new Date().toISOString() }),
        })
      )
    );

    res.status(201).json({ success: true, id: lead.id });
  } catch (error) {
    if (error instanceof z.ZodError) {
      return res.status(400).json({ error: 'Invalid input', details: error.errors });
    }
    console.error('Lead capture error:', error);
    res.status(500).json({ error: 'Failed to save lead. Please try again.' });
  }
});

leadsRouter.get('/', async (req, res) => {
  try {
    const allLeads = await db.select().from(leads).orderBy(leads.createdAt);
    res.json(allLeads);
  } catch (error) {
    res.status(500).json({ error: 'Failed to fetch leads' });
  }
});
```

### `src/routes/payments.ts` — Stripe Checkout + Webhooks

```typescript
import { Router } from 'express';
import Stripe from 'stripe';
import { z } from 'zod';
import { db } from '../db';
import { customers } from '../db/schema';

export const paymentsRouter = Router();
const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, { apiVersion: '2024-12-18.acacia' });

const checkoutSchema = z.object({
  priceId: z.string(),
  customerEmail: z.string().email(),
  successUrl: z.string().url(),
  cancelUrl: z.string().url(),
  mode: z.enum(['payment', 'subscription']),
});

paymentsRouter.post('/checkout', async (req, res) => {
  try {
    const { priceId, customerEmail, successUrl, cancelUrl, mode } = checkoutSchema.parse(req.body);
    const session = await stripe.checkout.sessions.create({
      payment_method_types: ['card'],
      customer_email: customerEmail,
      line_items: [{ price: priceId, quantity: 1 }],
      mode,
      success_url: successUrl,
      cancel_url: cancelUrl,
      metadata: { source: 'website' },
    });
    res.json({ url: session.url });
  } catch (error) {
    if (error instanceof z.ZodError) {
      return res.status(400).json({ error: 'Invalid input', details: error.errors });
    }
    res.status(500).json({ error: 'Failed to create checkout session' });
  }
});

paymentsRouter.post('/webhook', async (req, res) => {
  const sig = req.headers['stripe-signature']!;
  let event: Stripe.Event;
  try {
    event = stripe.webhooks.constructEvent(req.body, sig, process.env.STRIPE_WEBHOOK_SECRET!);
  } catch (err) {
    return res.status(400).json({ error: 'Invalid webhook signature' });
  }

  switch (event.type) {
    case 'checkout.session.completed': {
      const session = event.data.object as Stripe.CheckoutSession;
      await db.insert(customers).values({
        stripeCustomerId: session.customer as string,
        email: session.customer_email!,
        status: 'active',
      }).onConflictDoNothing();
      break;
    }
    case 'customer.subscription.deleted': {
      // Handle cancellation
      break;
    }
  }
  res.json({ received: true });
});
```

## Quality Rules

- Every endpoint returns: `{ success, data?, error? }`
- Never swallow errors — log them, return safe client message
- Lead schema columns must match Step 2 form fields exactly
- GHL + n8n fire in parallel via `Promise.allSettled` — never sequential
- Stripe webhook events must be handled — unhandled events cause reconciliation failures
